1、private修饰符

public class Person{
    private String name;
	private int mathScore;
	private int englishScore;
	private int amountScore;
	
	
	public void setName(String name){
	    this.name = name;	
	}
	......
	
	public void setAmountScore(){
	   this.amountScore = this.mathScore + this.englishScore;	
	}

}


main{
	
	Person p = new Person();	
	//p.amountScore   已经无法这样引用amountScore变量
	p.setName("宝宝");
	
	p.setMathScore(90);
	p.setEnglishScore(80);
	
	p.setAmountScore();
	
	
}
